########################################################################
# Check if there is C++ code at all
########################################################################
if(NOT tuto_sources)
    MESSAGE(STATUS "No C++ sources... skipping python bindings")
    return()
endif(NOT tuto_sources)

########################################################################
# Check for pygccxml
########################################################################
GR_PYTHON_CHECK_MODULE_RAW(
    "pygccxml"
    "import pygccxml"
    PYGCCXML_FOUND
    )

include(GrPybind)

########################################################################
# Python Bindings
########################################################################

list(APPEND tuto_python_files
    amp_python.cc
    mov_avg_python.cc python_bindings.cc)

GR_PYBIND_MAKE_OOT(tuto
   ../../..
   gr::tuto
   "${tuto_python_files}")

# copy bindings extension for use in QA test module
add_custom_command(TARGET tuto_python POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy $<TARGET_FILE:tuto_python>
    ${PROJECT_BINARY_DIR}/test_modules/gnuradio/tuto/
)

install(TARGETS tuto_python DESTINATION ${GR_PYTHON_DIR}/gnuradio/tuto COMPONENT pythonapi)
